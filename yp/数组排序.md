# 数组排序
## 插入排序

```
    var arr=[23,45,11,2,4,88,34,100];
    var i=j=temp=0;
    for(i=1;i<arr.length;i++){
        temp=arr[i];    //i=j+1,待排序
        for(j=i-1;j>=0;j--){     //j=1-1,已排序
            if(arr[j]>temp){    //已排序大于未排序，未排序前插(与已排序交换位置)
                arr[j+1]=arr[j];
            }else{
                break;  //如果待排序比已排序大，直接break
            }
        }
        arr[j+1]=temp;  //break后待排序插入已排序后面
    }
    console.log(arr);
```
